<div class="modal" id="editModal">

    <input type="text" id="editInput" placeholder="할 일 제목">

    <select id="editPriority">

      <option value="high">🔴 높음</option>

      <option value="mid">🟡 중간</option>

      <option value="low">🟢 낮음</option>

    </select>

    <input type="date" id="editDueDate">

    <div class="modal-buttons">

      <button class="btn-cancel" onclick="TodoController.closeModal()">취소</button>

      <button class="btn-save" onclick="TodoController.saveEdit()">저장</button>

<script>

  // ── Model ──────────────────────────────

  const TodoModel = {

    todos: JSON.parse(localStorage.getItem('todos') || '[]'),


    save() {

      localStorage.setItem('todos', JSON.stringify(this.todos));

    },


    add(title, priority, dueDate) {

      const todo = {

        id: Date.now(),

        title,

        priority,

        dueDate,

        completed: false,

        createdAt: new Date().toISOString()

      };

      this.todos.push(todo);

      this.save();

    },


    delete(id) {

      this.todos = this.todos.filter(t => t.id !== id);

      this.save();

    },


    toggle(id) {

      const todo = this.todos.find(t => t.id === id);

      if (todo) { todo.completed = !todo.completed; this.save(); }

    },


    update(id, title, priority, dueDate) {

      const todo = this.todos.find(t => t.id === id);

      if (todo) {

        todo.title = title;

        todo.priority = priority;

        todo.dueDate = dueDate;

        this.save();

      }

    },


    filter(type) {

      if (type === 'active') return this.todos.filter(t => !t.completed);

      if (type === 'completed') return this.todos.filter(t => t.completed);

      return this.todos;

    },


    isOverdue(dueDate) {

      if (!dueDate) return false;

      return new Date(dueDate) < new Date(new Date().toDateString());

    }

  };


  // ── View ───────────────────────────────

  const TodoView = {

    currentFilter: 'all',


    render() {

      const list = document.getElementById('todoList');

      const todos = TodoModel.filter(this.currentFilter);


      if (todos.length === 0) {

        list.innerHTML = '<div class="empty">할 일이 없어요 🎉</div>';

        return;

      }


      list.innerHTML = todos.map(todo => {

        const overdue = !todo.completed && TodoModel.isOverdue(todo.dueDate);

        return `

          <div class="todo-item ${todo.completed ? 'completed' : ''} ${overdue ? 'overdue' : ''}">

            <input type="checkbox" ${todo.completed ? 'checked' : ''}

              onchange="TodoController.toggleTodo(${todo.id})">

            <span class="todo-title">${todo.title}</span>

            <span class="badge ${todo.priority}">

              ${todo.priority === 'high' ? '높음' : todo.priority === 'mid' ? '중간' : '낮음'}

            </span>

            ${todo.dueDate ? `<span class="due-date ${overdue ? 'overdue' : ''}">📅 ${todo.dueDate} ${overdue ? '⚠️ 마감 초과' : ''}</span>` : ''}

            <button class="btn-edit" onclick="TodoController.openEdit(${todo.id})">수정</button>

            <button class="btn-delete" onclick="TodoController.deleteTodo(${todo.id})">삭제</button>

          </div>

        `;

      }).join('');

    }

  };


  // ── Controller ─────────────────────────

  const TodoController = {

    editingId: null,


    addTodo() {

      const title = document.getElementById('todoInput').value.trim();

      if (!title) { alert('할 일을 입력해주세요!'); return; }

      const priority = document.getElementById('prioritySelect').value;

      const dueDate = document.getElementById('dueDateInput').value;

      TodoModel.add(title, priority, dueDate);

      document.getElementById('todoInput').value = '';

      document.getElementById('dueDateInput').value = '';

      TodoView.render();

    },


    deleteTodo(id) {

      if (confirm('정말 삭제할까요?')) {

        TodoModel.delete(id);

        TodoView.render();

      }

    },


    toggleTodo(id) {

      TodoModel.toggle(id);

      TodoView.render();

    },


    filterTodos(type, btn) {

      document.querySelectorAll('.filter-box button').forEach(b => b.classList.remove('active'));

      btn.classList.add('active');

      TodoView.currentFilter = type;

      TodoView.render();

    },


    openEdit(id) {

      const todo = TodoModel.todos.find(t => t.id === id);

      if (!todo) return;

      this.editingId = id;

      document.getElementById('editInput').value = todo.title;

      document.getElementById('editPriority').value = todo.priority;

      document.getElementById('editDueDate').value = todo.dueDate || '';

      document.getElementById('editModal').classList.add('show');

    },


    closeModal() {

      document.getElementById('editModal').classList.remove('show');

      this.editingId = null;

    },


    saveEdit() {

      const title = document.getElementById('editInput').value.trim();

      if (!title) { alert('할 일을 입력해주세요!'); return; }

      const priority = document.getElementById('editPriority').value;

      const dueDate = document.getElementById('editDueDate').value;

      TodoModel.update(this.editingId, title, priority, dueDate);

      this.closeModal();

      TodoView.render();

    }

  };


  TodoView.render();

</script>
